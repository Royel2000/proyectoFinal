// alumno.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root',
})
export class AlumnoService {
  private apiUrl = 'tu_url_del_api'; // Reemplaza esto con tu URL real

  constructor(private http: HttpClient) {}

  obtenerAlumnos(): Observable<Alumno[]> {
    return this.http.get<Alumno[]>(`${this.apiUrl}/alumnos`);
  }
}



// home.component.ts
import { Component, OnInit } from '@angular/core';
import { AlumnoService } from 'path/to/alumno.service'; // Reemplaza esto con la ruta correcta
import { Alumno } from 'path/to/alumno.model'; // Reemplaza esto con la ruta correcta

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.css'],
})
export class HomeComponent implements OnInit {
  alumnos: Alumno[] = [];

  constructor(private alumnoService: AlumnoService) {}

  ngOnInit() {
    this.obtenerAlumnos();
  }

  obtenerAlumnos() {
    this.alumnoService.obtenerAlumnos().subscribe(
      (alumnos) => {
        this.alumnos = alumnos;
      },
      (error) => {
        console.error('Error al obtener alumnos:', error);
      }
    );
  }
}
